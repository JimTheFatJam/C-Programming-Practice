  4  struct Node {               // Node structure for linked list
  5      int data;               // Data field can store any type of data and any value
  6      struct Node* next;      // Node pointer that links to the next node
  7  };
  8  
  9  struct Node* createNode(int new_data) {                                     // Function to create new node in linked list
 10      struct Node* new_node = (struct Node*)malloc(sizeof(struct Node));      // Allocate memory for new node
 11  
 12      if(new_node == NULL) {                              // Check is memory allocation was successful
 13          printf("\nMemory allocation failed!\n");        // Print error message
 14          exit(EXIT_FAILURE);                             // Exit program
 15      }
 16  
 17      new_node->data = new_data;      // Assign the given data to the new node's data field
 18      new_node->next = NULL;          // Initialize the next pointer to NULL
 19      return new_node;                // Return the pointer to the newly created node
 20  };
 21  
 22  void printList(struct Node* head) {     // Function to print the elements of a linked list
 23      struct Node* curr = head;           // Create a temporary pointer to traverse the list,
 24                                          // starting at the head
 25  
 26      while (curr != NULL) {              // Traverse the linked list until NULL is reached
 27          printf("%d->", curr->data);     // Print the current node's data
 28          curr = curr->next;              // Move to the next node in the list
 29      }
 30      printf("NULL\n");                   // Print "NULL" to indicate end
 31  }

 33  void insertAtFront(struct Node** head, int new_data) {  // Function to insert a new node at the front of the linked list
 34      struct Node* new_node = createNode(new_data);       // Create new node with given data
 35      new_node->next = *head;                             // Make the next pointer of the new node point to the current head
 36      *head = new_node;                                   // Update the head pointer to point to the new node,
 37  }                                                       // making it the new head
 38  
 39  void append(struct Node** head, int new_data) {         // Function to insert a new node at the end of the linked list
 40      struct Node* new_node = createNode(new_data);       // Create new node with given data
 41  
 42      if(*head == NULL) {         // If list is empty, make the new node the head
 43          *head = new_node;       // Assign the new node as the head
 44          return;                 // Exit function
 45      }
 46  
 47      struct Node* last = *head;
 48      while(last->next != NULL) {     // Traverse the linked list until the last node
 49          last = last->next;          // Move to the next node
 50      }
 51  
 52      last->next = new_node;           // Change the next pointer of the last node to point to the new node
 53  }

 55  void insertPos(struct Node** head, int pos, int new_data) {     // Function to insert a new node at a specific
 56                                                                  // position in the linked list
 57      if(pos < 1) return;                                     // Return if the given position is invalid (less than 1)
 58  
 59      struct Node* new_node = createNode(new_data);           // Create a new node with the given data
 60  
 61      if(pos == 1) {                      // If inserting at position 1, update the head of the list (insertAtFront)
 62          new_node->next = *head;         // Make the next pointer of the new node point to the current head
 63          *head = new_node;               // Update the head pointer to point to the new node, making it the new head
 64          return;
 65      }
 66  
 67      struct Node* curr = *head;
 68      for(int i = 1; i < pos - 1 && curr != NULL; i++) {      // Traverse to the node that will be just before the new node
 69          curr = curr->next;                                  // Move to the next node
 70      }
 71  
 72      if(curr == NULL) return;            // If the position is greater than the number of nodes, return without inserting
 73  
 74      new_node->next = curr->next;        // Update the next pointers to insert the new node at the correct position
 75      curr->next = new_node;
 76  }

 78  void deleteHead(struct Node** head) {       // Function to delete the node at the front of the linked list
 79      if(*head == NULL) return;               // Return if list is empty (nothing to delete)
 80  
 81      struct Node* temp = *head;              // Store the current head node in a temporary variable
 82      *head = (*head)->next;                  // Move the head pointer to the next node in the list
 83  
 84      free(temp);                             // Free the memory allocated for the old head node
 85  }
 86  
 87  void deleteEnd(struct Node** head) {        // Function to delete the node at the end of the linked list
 88      if(*head == NULL) return;               // Return if list is empty (nothing to delete)
 89  
 90      if((*head)->next == NULL) {             // If only one node in the list
 91          free(*head);                        // Free the only node
 92          *head = NULL;                       // Set head to NULL since the list is now empty
 93          return;
 94      }
 95  
 96      struct Node* second_last = *head;
 97      while(second_last->next->next != NULL) {    // Traverse to the second last node
 98          second_last = second_last->next;        // Move to the next node
 99      }
100  
101      free(second_last->next);                    // Free memory of the last node (node after second_last)
102      second_last->next = NULL;                   // Change the next pointer of the second-to-last node to NULL
103  }

105  void deletePos(struct Node** head, int pos) {       // Function to delete a node at a given position in the linked list
106      if(*head == NULL) return;                       // Return if list is empty (nothing to delete)
107  
108      struct Node* temp = *head;                  // Temporary pointer to traverse the list
109      struct Node* prev = NULL;                   // Pointer to store the previous node
110  
111      if(pos == 1) {                  // If the node to be deleted is the head (position 1)
112          *head = temp->next;         // Move the head pointer to the next node
113          free(temp);                 // Free memory of the old head node
114          return;
115      }
116  
117      for(int i = 1; i < pos && temp != NULL; i++) {      // Traverse the list to find the node at the given position
118          prev = temp;                                    // Store the previous node
119          temp = temp->next;                              // Move to the next node
120      }
121  
122      if(temp == NULL) return;        // If the position is greater than the number of nodes, return
123  
124      prev->next = temp->next;        // Update the next pointer of the previous node to skip the node to be deleted
125      free(temp);                     // Free memory of the node to be deleted
126  }

128  int getLength(struct Node* head) {      // Function to get the number of nodes of linked list
129      int len = 0;                        // Initialize a counter 'len' to store the length of the list
130  
131      struct Node* curr = head;
132      while(curr != NULL) {               // Traverse the linked list until NULL is reached
133          len++;                          // Increment the length counter for each node
134          curr = curr->next;              // Move to the next node
135      }
136  
137      return len;                         // Return the total number of nodes in the list
138  }

140  void bubbleSort(struct Node** head) {           // Function to sort linked list with bubble sort
141      int swapped;                                // Flag to check if any swapping was done in an iteration
142      struct Node* curr = NULL;                   // Pointer to track the current node
143      struct Node* last_sorted =  NULL;           // Marks the last sorted node (boundary for comparisons)
144  
145      if(*head == NULL || (*head)->next == NULL)   // Return if the list is empty or if only one node in the list
146          return;
147  
148      do {
149          swapped = 0;            // Reset swapped flag at the start of each iteration
150          curr = *head;           // Start traversing from the head
151  
152          while(curr->next != last_sorted) {                  // Traverse the list until the last sorted node
153  
154              if (curr->data > curr->next->data) {        // Compare adjacent nodes
155                  swapData(curr, curr->next);             // Swap if out of order
156                  swapped = 1;                            // Set swapped flag to indicate a change
157              }
158  
159              curr = curr->next;    // Move to the next node
160          }
161  
162          last_sorted = curr;     // Move the last sorted boundary one step back
163      } while(swapped);           // Continue until no swaps occur
164  }

166  void swapData(struct Node* nodeA, struct Node* nodeB) {         // Function to swap the data of two nodes
167      int temp = nodeA->data;                 // Store nodeA's data in a temporary variable
168      nodeA->data = nodeB->data;              // Assign nodeB's data to nodeA
169      nodeB->data = temp;                     // Assign temp (original nodeA data) to nodeB
170  }
